public class Animal{

private String raza;
private String  nombre;
private int edad;

 public Animal(String raza){
   raza=raza;
  }
 public Animal(String nombre){
   nombre=nombre;
  }
 public Animal(int edad){
   edad=edad;
  }
public void setAnimal(String raza){
   raza=raza;

}
public void setAnimal(String nombre){
   nombre=nombre;

}
public void setAnimal(int edad ){
   edad=edad;

}
public String getraza(){
   return raza;

}
public String getnombre(){
   return nombre;

}
public int getedad(){
   return edad;

}
